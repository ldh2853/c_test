구조체를 활용해서 2명의 이름과 성적을 초기화하고, 과목평균과 학생 평균을 화면애 출력하는 코드
```
#include <stdio.h>

#define NUM_STUDENTS 2
#define NUM_SUBJECTS 3

typedef struct {
    char name[20];
    int  scores[NUM_SUBJECTS];
} Student;

int main(void) {
    char* subjects[NUM_SUBJECTS] = { "수학", "영어", "과학" };

    Student students[NUM_STUDENTS] = {
        { "홍길동", { 90, 80, 70 } },
        { "김철수", { 85, 95, 75 } }
    };

    printf("=== 성적표 ===\n\n");

    // 학생별 평균
    printf("[ 학생 평균 ]\n");
    for (int i = 0; i < NUM_STUDENTS; i++) {
        int sum = 0;
        for (int j = 0; j < NUM_SUBJECTS; j++)
            sum += students[i].scores[j];
        printf("  %s : %.2f점\n", students[i].name, (double)sum / NUM_SUBJECTS);
    }

    // 과목별 평균
    printf("\n[ 과목 평균 ]\n");
    for (int j = 0; j < NUM_SUBJECTS; j++) {
        int sum = 0;
        for (int i = 0; i < NUM_STUDENTS; i++)
            sum += students[i].scores[j];
        printf("  %s : %.2f점\n", subjects[j], (double)sum / NUM_STUDENTS);
    }

    return 0;
}
```
