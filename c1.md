// breakout.c - 무지개 블록깨기 (raylib)
#include "raylib.h"
#include <stdio.h>
#include <stdbool.h>
#include <math.h>

#define SCREEN_W      800
#define SCREEN_H      600
#define MAX_LIVES     5

#define BLOCK_COLS    10
#define BLOCK_ROWS    6
#define BLOCK_W       64
#define BLOCK_H       22
#define BLOCK_PAD_X   ((SCREEN_W - BLOCK_COLS * BLOCK_W) / 2)
#define BLOCK_PAD_Y   60

#define PADDLE_W      110
#define PADDLE_H      14
#define PADDLE_Y      (SCREEN_H - 60)
#define PADDLE_SPEED  600

#define BALL_R        9
#define BALL_SPEED    380

typedef struct {
    Rectangle rect;
    bool      alive;
    Color     color;
} Block;

typedef enum { STATE_PLAY, STATE_WIN, STATE_OVER } GameState;

static const Color ROW_COLORS[BLOCK_ROWS] = {
    { 255,  80,  80, 255 },
    { 255, 165,  40, 255 },
    { 255, 230,  40, 255 },
    {  60, 210,  80, 255 },
    {  60, 130, 255, 255 },
    { 160,  60, 255, 255 },
};

static Block     blocks[BLOCK_ROWS][BLOCK_COLS];
static Rectangle paddle;
static Vector2   ball;
static Vector2   ballVel;
static int       score;
static int       lives;
static GameState state;

static void InitGame(void) {
    for (int r = 0; r < BLOCK_ROWS; r++) {
        for (int c = 0; c < BLOCK_COLS; c++) {
            blocks[r][c].rect = (Rectangle){
                (float)(BLOCK_PAD_X + c * BLOCK_W + 2),
                (float)(BLOCK_PAD_Y + r * (BLOCK_H + 6)),
                (float)(BLOCK_W - 4),
                (float)BLOCK_H
            };
            blocks[r][c].alive = true;
            blocks[r][c].color = ROW_COLORS[r];
        }
    }
    paddle  = (Rectangle){ (float)((SCREEN_W - PADDLE_W) / 2), (float)PADDLE_Y, (float)PADDLE_W, (float)PADDLE_H };
    ball    = (Vector2){ (float)(SCREEN_W / 2), (float)(SCREEN_H / 2 + 80) };
    ballVel = (Vector2){ BALL_SPEED * 0.6f, -(float)BALL_SPEED };
    score   = 0;
    lives   = MAX_LIVES;
    state   = STATE_PLAY;
}

static bool CollideBallRect(Rectangle rect) {
    float cx = ball.x, cy = ball.y;
    float nearX = cx < rect.x ? rect.x : (cx > rect.x + rect.width  ? rect.x + rect.width  : cx);
    float nearY = cy < rect.y ? rect.y : (cy > rect.y + rect.height ? rect.y + rect.height : cy);
    float dx = cx - nearX, dy = cy - nearY;
    if (dx * dx + dy * dy >= (float)(BALL_R * BALL_R)) return false;
    float overlapX = (float)BALL_R - fabsf(dx);
    float overlapY = (float)BALL_R - fabsf(dy);
    if (overlapX < overlapY)
        ballVel.x = (dx >= 0) ? fabsf(ballVel.x) : -fabsf(ballVel.x);
    else
        ballVel.y = (dy >= 0) ? fabsf(ballVel.y) : -fabsf(ballVel.y);
    return true;
}

static void UpdateGame(float dt) {
    if (state != STATE_PLAY) return;

    if (IsKeyDown(KEY_LEFT)  || IsKeyDown(KEY_A)) paddle.x -= PADDLE_SPEED * dt;
    if (IsKeyDown(KEY_RIGHT) || IsKeyDown(KEY_D)) paddle.x += PADDLE_SPEED * dt;
    if (paddle.x < 0) paddle.x = 0;
    if (paddle.x + PADDLE_W > SCREEN_W) paddle.x = (float)(SCREEN_W - PADDLE_W);

    ball.x += ballVel.x * dt;
    ball.y += ballVel.y * dt;

    if (ball.x - BALL_R <= 0)        { ball.x = (float)BALL_R;            ballVel.x =  fabsf(ballVel.x); }
    if (ball.x + BALL_R >= SCREEN_W) { ball.x = (float)(SCREEN_W - BALL_R); ballVel.x = -fabsf(ballVel.x); }
    if (ball.y - BALL_R <= 0)        { ball.y = (float)BALL_R;            ballVel.y =  fabsf(ballVel.y); }

    if (ball.y - BALL_R > SCREEN_H) {
        if (--lives <= 0) { state = STATE_OVER; return; }
        ball    = (Vector2){ (float)(SCREEN_W / 2), (float)(SCREEN_H / 2 + 80) };
        ballVel = (Vector2){ BALL_SPEED * 0.6f, -(float)BALL_SPEED };
    }

    if (CollideBallRect(paddle)) {
        float hitPos = (ball.x - paddle.x) / PADDLE_W;
        ballVel.x = (hitPos - 0.5f) * 2.0f * BALL_SPEED;
        ballVel.y = -fabsf(ballVel.y);
    }

    int remaining = 0;
    for (int r = 0; r < BLOCK_ROWS; r++) {
        for (int c = 0; c < BLOCK_COLS; c++) {
            if (!blocks[r][c].alive) continue;
            remaining++;
            if (CollideBallRect(blocks[r][c].rect)) {
                blocks[r][c].alive = false;
                score += (BLOCK_ROWS - r) * 10;
                remaining--;
            }
        }
    }
    if (remaining == 0) state = STATE_WIN;
}

static void DrawGame(void) {
    BeginDrawing();
    ClearBackground((Color){ 18, 18, 28, 255 });

    for (int r = 0; r < BLOCK_ROWS; r++) {
        for (int c = 0; c < BLOCK_COLS; c++) {
            if (!blocks[r][c].alive) continue;
            Rectangle br = blocks[r][c].rect;
            DrawRectangleRec(br, blocks[r][c].color);
            DrawRectangle((int)br.x, (int)br.y, (int)br.width, 4, (Color){ 255, 255, 255, 70 });
            DrawRectangleLinesEx(br, 1, (Color){ 0, 0, 0, 120 });
        }
    }

    DrawRectangleRounded(paddle, 0.4f, 8, (Color){ 200, 220, 255, 255 });
    DrawRectangleRounded(
        (Rectangle){ paddle.x + 4, paddle.y + 2, paddle.width - 8, (float)(PADDLE_H / 2) },
        0.5f, 8, (Color){ 255, 255, 255, 100 });

    DrawCircleV(ball, (float)(BALL_R + 4), (Color){ 255, 255, 150, 50 });
    DrawCircleV(ball, (float)BALL_R,       (Color){ 255, 240, 120, 255 });
    DrawCircleV(ball, (float)(BALL_R - 3), (Color){ 255, 255, 220, 180 });

    char scoreText[32];
    sprintf_s(scoreText, sizeof(scoreText), "SCORE  %d", score);
    DrawText(scoreText, 20, 14, 22, (Color){ 200, 200, 255, 255 });

    DrawText("LIVES", SCREEN_W - 160, 14, 20, (Color){ 200, 200, 255, 255 });
    for (int i = 0; i < MAX_LIVES; i++) {
        Color c = (i < lives) ? (Color){ 255, 80, 100, 255 } : (Color){ 80, 80, 100, 255 };
        DrawCircle(SCREEN_W - 70 + i * 18, 24, 7, c);
    }

    DrawLine(0, 50, SCREEN_W, 50, (Color){ 80, 80, 120, 180 });

    if (state == STATE_WIN) {
        DrawRectangle(0, 0, SCREEN_W, SCREEN_H, (Color){ 0, 0, 0, 140 });
        DrawText("YOU WIN!", 220, 220, 72, YELLOW);
        DrawText("Press R to restart", 260, 310, 28, LIGHTGRAY);
        char finalScore[32];
        sprintf_s(finalScore, sizeof(finalScore), "Final Score: %d", score);
        DrawText(finalScore, 280, 360, 26, WHITE);
    }
    if (state == STATE_OVER) {
        DrawRectangle(0, 0, SCREEN_W, SCREEN_H, (Color){ 0, 0, 0, 160 });
        DrawText("GAME OVER", 200, 220, 72, RED);
        DrawText("Press R to restart", 260, 310, 28, LIGHTGRAY);
        char finalScore[32];
        sprintf_s(finalScore, sizeof(finalScore), "Final Score: %d", score);
        DrawText(finalScore, 280, 360, 26, WHITE);
    }

    EndDrawing();
}

int main(void) {
    InitWindow(SCREEN_W, SCREEN_H, "Rainbow Breakout");
    SetTargetFPS(60);
    InitGame();

    while (!WindowShouldClose()) {
        float dt = GetFrameTime();
        if ((state == STATE_WIN || state == STATE_OVER) && IsKeyPressed(KEY_R))
            InitGame();
        UpdateGame(dt);
        DrawGame();
    }

    CloseWindow();
    return 0;
}
